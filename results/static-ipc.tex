\pgfplotsset{
  mylineO/.style={line width=1.2pt, mark size=2.5pt, color=tabblue, mark=o},
  mylineX/.style={line width=1.2pt,  mark size=2.5pt, color=taborange, mark=x},
  mylinePlus/.style={line width=1.2pt, mark size=2.5pt, color=tabgreen, mark=+},
  mylineCircle/.style={line width=1.2pt, mark size=1.5pt, color=tabpurple, mark=*},
  mylineSquare/.style={line width=1.2pt, mark size=1.5pt, color=tabbrown, mark=square*},
  mylineStar/.style={line width=1.2pt, mark size=2.5pt, color=tabolive, mark=star},
  mylineTri/.style={line width=1.2pt, mark size=1.5pt, color=tabgray, mark=triangle*},
 only if/.style args={entry of #1 is #2}{
    x filter/.code={
      \edef\tempa{\thisrow{#1}}
      \edef\tempb{#2}
      \ifx\tempa\tempb
      \else
        \def\pgfmathresult{}
      \fi
    }
  }
}
